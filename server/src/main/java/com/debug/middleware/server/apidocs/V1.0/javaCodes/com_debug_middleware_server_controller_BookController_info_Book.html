public class Book implements Serializable{

    private int bookNo; //图书编号
    private String name; //图书名称

    public int getBookNo(){
        return bookNo;
    }

    public void setBookNo(int bookNo){
        this.bookNo = bookNo;
    }

    public String getName(){
        return name;
    }

    public void setName(String name){
        this.name = name;
    }

}
